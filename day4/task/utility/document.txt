// create users table

CREATE TABLE `users` (
  `user_id` int NOT NULL AUTO_INCREMENT,
  `name` varchar(50) NOT NULL,
  `email` varchar(50) NOT NULL,
  `password` varchar(255) NOT NULL,
  `register_at` timestamp NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`user_id`),
  UNIQUE KEY `email` (`email`)
)



// create accounts table
CREATE TABLE `accounts` (
  `acc_id` int NOT NULL AUTO_INCREMENT,
  `balance` decimal(10,2) NOT NULL,
  `user_id` int DEFAULT NULL,
  `update_at` timestamp NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`acc_id`),
  KEY `user_id` (`user_id`),
  CONSTRAINT `accounts_ibfk_1` FOREIGN KEY (`user_id`) REFERENCES `users` (`user_id`) ON DELETE CASCADE
) 


// creating index 

CREATE INDEX user_id_index ON accounts(user_id);




// trigger for balance update


DELIMITER //

CREATE TRIGGER user_balance_update
BEFORE UPDATE ON accounts
FOR EACH ROW
BEGIN
    SET NEW.update_at = CURRENT_TIMESTAMP;
END //

DELIMITER ;



// transction table 

CREATE TABLE transactions(
	transaction_id INT PRIMARY KEY NOT NULL,
    transaction_type VARCHAR(50) NOT NULL,
    transaction_amount VARCHAR(50) NOT NULL,
    acc_id INT,
    user_id INT,
    update_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    order_id INT,
    FOREIGN KEY (acc_id) REFERENCES accounts (acc_id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users (user_id) ON DELETE CASCADE,
    FOREIGN KEY (order_id) REFERENCES product_order (order_id) ON DELETE CASCADE
);


DELIMITER //

CREATE TRIGGER transaction_balance_update
AFTER INSERT ON transactions
FOR EACH ROW
BEGIN

    IF NEW.transaction_type = 'credit' THEN
        UPDATE accounts
        SET balance = balance + NEW.transaction_amount
        WHERE acc_id = NEW.acc_id;
    END IF;

    IF NEW.transaction_type = 'debit' THEN
        UPDATE accounts
        SET balance = balance - NEW.transaction_amount
        WHERE acc_id = NEW.acc_id;
    END IF;

END//

DELIMITER ;

// product table
CREATE TABLE products(
	product_id INT PRIMARY KEY AUTO_INCREMENT,
    product_name VARCHAR(50),
    product_detail VARCHAR(255),
    product_image VARCHAR(255),
    create_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    user_id INT NOT NULL,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE
);

// order table
CREATE TABLE product_order(
	order_id INT PRIMARY KEY AUTO_INCREMENT,
    order_qty INT,
    order_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    total_amount DECIMAL(10,2),
    product_id INT,
    user_id INT,
    FOREIGN KEY (product_id) REFERENCES products (product_id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users (user_id) ON DELETE CASCADE
);



// cart table

CREATE TABLE cart (
	cart_id INT PRIMARY KEY AUTO_INCREMENT,
    product_id INT ,
    user_id INT,
    FOREIGN KEY (product_id) REFERENCES products (product_id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users (user_id) ON DELETE CASCADE
);



// join query


SELECT c.cart_id, c.user_id, p.product_id, p.product_name, p.product_detail, p.product_price, p.product_image
FROM cart c
JOIN products p ON c.product_id = p.product_id WHERE c.user_id = 9
ORDER BY c.cart_id ASC ;
